def compute_percentage_change(mean_viewership):
    percentage_changes = []
    for i in range(1, len(mean_viewership)):
        change = ((mean_viewership[i] - mean_viewership[i - 1]) / mean_viewership[i - 1]) * 100
        percentage_changes.append(change)
    return percentage_changes

mean_viewership = df.groupby('Season')['Viewership'].mean().reset_index()
percentage_changes = compute_percentage_change(mean_viewership['Viewership'])

plt.figure(figsize=(10, 6))
sns.lineplot(data=mean_viewership, x='Season', y='Viewership', marker='o', linestyle='-')
for i, change in enumerate(percentage_changes):
    plt.text(i+2, mean_viewership['Viewership'][i+1], f'{change:.2f}%', ha='right', va='bottom')
plt.title('Viewership season to season')
plt.xlabel('Season')
plt.ylabel('Mean US viewership (millions)')
plt.grid(True)
plt.tight_layout()
plt.show()
